            const titles = [

            let titleIndex = 0;
            const titleElement = document.querySelector('.typing-text');
            
            function typeTitle() {
                const currentTitle = titles[titleIndex];
                titleElement.textContent = '';
                let charIndex = 0;
                
                const typingInterval = setInterval(() => {
                    titleElement.textContent += currentTitle[charIndex];
                    charIndex++;
                    
                    if (charIndex === currentTitle.length) {
                        clearInterval(typingInterval);
                        setTimeout(() => {
                            deleteTitle();
                        }, 2000);
                    }
                }, 100);
            }
            
            function deleteTitle() {
                const currentText = titleElement.textContent;
                let charIndex = currentText.length - 1;
                
                const deletingInterval = setInterval(() => {
                    titleElement.textContent = currentText.substring(0, charIndex);
                    charIndex--;
                    
                    if (charIndex < 0) {
                        clearInterval(deletingInterval);
                        titleIndex = (titleIndex + 1) % titles.length;
                        setTimeout(typeTitle, 500);
                    }
                }, 50);
            }
            
            // Start the typing effect
            typeTitle();
            
            // Terminal blinking cursor effect
            const terminal = document.querySelector('.terminal');
            setInterval(() => {
                const blinkElement = terminal.querySelector('.blink');
                if (blinkElement) {
                    blinkElement.style.visibility = blinkElement.style.visibility === 'hidden' ? 'visible' : 'hidden';
                }
            }, 500);
        });
